# 计算 edQTL nominal output 中 Beta (COLUMN_14) 的标准误差 (SE)
# 基于 P 值 (COLUMN_12)

#suppressMessages(library(argparser))
suppressMessages(library(data.table))
suppressMessages(library(dplyr))

#input_file <- "/gpfs/hpc/home/chenchao/mayinuo/edqtl/meta/metaqtl/nominal_all/edqtl_nominal_EUR_final_cols.txt"
#output_file <- "/gpfs/hpc/home/chenchao/mayinuo/edqtl/meta/metaqtl/nominal_all/edqtl_nominal_EUR_with_SE.txt" 

input_file <- "/gpfs/hpc/home/chenchao/mayinuo/edqtl/meta/metaqtl/nominal_all/edqtl_nominal_EAS_final_cols.txt"
output_file <- "/gpfs/hpc/home/chenchao/mayinuo/edqtl/meta/metaqtl/nominal_all/edqtl_nominal_EAS_with_SE.txt" 


# 2. 读取数据
cat("Reading input file:", input_file, "\n")
dat <- fread(input_file, data.table = FALSE, header = TRUE)



# 检查文件是否为空
if (nrow(dat) == 0) {
    stop("Input file is empty. Exiting.")
}

# 3. 定义关键列名
# 根据您提供的文件头信息，FastQTL 结果中的 P 值和 Beta/Slope 位于以下列
P_COL_NAME <- "COLUMN_12"
BETA_COL_NAME <- "COLUMN_14"

# 4. 检查关键列是否存在
if (!all(c(P_COL_NAME, BETA_COL_NAME) %in% colnames(dat))) {
    stop(paste("Required columns", P_COL_NAME, "and", BETA_COL_NAME, "not found in the input file."))
}

# 5. 计算 Z-score 和 SE
cat("Calculating SE based on P-value (", P_COL_NAME, ") and Beta (", BETA_COL_NAME, ").\n")

# FastQTL/QTL 通常提供双侧 P 值。根据 Beta 的符号确定 Z-score 的符号。
# 核心逻辑:
# Z = qnorm(P_value / 2) if Beta < 0
# Z = qnorm(1 - P_value / 2) if Beta > 0
# SE = Beta / Z

dat <- dat %>%
    mutate(
        # 获取 P 值
        P_value = as.numeric(!!sym(P_COL_NAME)), 
        # 获取 Beta
        Beta = as.numeric(!!sym(BETA_COL_NAME)),
        
        # 计算 Z-score
        Z_score = case_when(
            # 如果 Beta > 0 (正效应), Z-score 位于分布的右侧
            Beta > 0 ~ qnorm(1 - P_value / 2),
            # 如果 Beta < 0 (负效应), Z-score 位于分布的左侧
            Beta < 0 ~ qnorm(P_value / 2),
            # 如果 Beta = 0 (或 P-value 接近 1), Z-score 接近 0
            TRUE ~ 0
        ),
        
        # 计算 SE: SE = Beta / Z_score
        # 避免除以 0 (当 P-value = 1 或 Z-score = 0 时)
        SE = case_when(
            abs(Z_score) > 1e-10 ~ abs(Beta) / abs(Z_score),
            TRUE ~ NA_real_ # 如果 Z-score 接近 0, SE 设为 NA
        )
    )

# 6. 选择原始列和新的 SE 列进行输出
dat_output <- dat %>%
    # 保持所有原始列
    select(everything(), SE) %>%
    # 移除临时的 P_value, Beta, Z_score 列 (如果不需要)
    select(-P_value, -Beta, -Z_score)

# 7. 输出结果
cat("Writing output file:", output_file, "\n")
# 注意：使用 output_file 变量
write.table(dat_output, output_file, col.names = TRUE, row.names = FALSE, quote = FALSE, sep = "\t")

cat("SE calculation complete. The new column 'SE' has been added to the output file.\n")
